附錄C: 用C來實作物件導向的概念

 

近年來許多專家學者的研究指出：在嵌入式系統領域中採用物件導向的觀念進行設計有很多優點；不過實務上大部分嵌入式系統開發仍然以採用C語言為主，主因是有的平台沒有C++編譯器可用，有的產品則CPU運算效能比較差，無法承受C++所帶來效能變慢的風險；而且C++編譯器產生的code較無法掌握，始終是其應用在嵌入式系統領域的致命傷。

 

可是使用物件導向觀念進行系統設計的好處卻總令人不忍割捨；所以實務上通常還是會使用物件導向方法論進行系統設計，實作時則使用C語言來模擬物件導向中的各種觀念。其實只要不是用到物件導向中太複雜的概念，使用C語言還是可以很優雅的實現物件導向的系統。

 

這樣的作法有一個基本的觀念支持：基本上嵌入式系統是一種封閉式系統，也就是說所有可能在這個系統上進行開發工作的工程師應該都是“自己人”，亦即所有的開發動作都應該以設計文件為依歸，例如：當設計文件規定某個資料結構裡的某個屬性乃屬於私有屬性(private attribute)，則除了該模組內的函數可以操作它之外，其他的函數在使用該資料結構時，其他的函數性的件為依歸只能透過該模組公開的API來間接操作這些私有屬性。若某工程師違反了此一原則，雖然該程式還是可以編譯通過並正常運行，但其違反的不是程式語言的規範，而是該專案團隊的規定，照理說當然應該要嚴格管制。

 

也就是說，C的語法雖然無法完美模擬物件導向的所有觀念，但是嵌入式系統團隊卻可藉由系統設計文件來規範工程師，從而使物件導向的系統可以用C語言來實現。

 

以下我們用一個實例來說明如何用C實現物件導向的各種概念，讀者只要詳讀程式中的註解(粗體字部分)，應該就可以了解這個方法的精神：e譯器產生的者醒你寫程式

 

C-1 屬性(attribute)

 

類別定義是用資料結構的方式來模擬，但有兩個缺點：一個是無法區分私有屬性(private)或公有屬性(public)；第二個是類別屬性與方法無法強制關聯。

 

在my_object.h中除了定義物件的公用與私有屬性外，公用方法(public method)與類別屬性(class attribute)也必須在此宣告；則其他程式只要include這個.h檔，即可操作這個物件。e譯器產生的者醒你寫程式

 

 

/******************************************************

            Filename: my_object.h

類別定義 – 屬性與公用方法(public method)宣告

******************************************************/

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

// 類別屬性 : 屬性定義

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

 

struct my_object

{

    // 所有類別都有的屬性, 用以紀錄該類別基本的屬性

    // 這個屬性必須是所有類別屬性的第一個,

// 沒有這個屬性就無法實現Polymorphism (本節稍後會解釋)

    //

    OBJECT base ;

   

    // 物件公開屬性

    //

    int public_att1 ;

    int public_att2 ;

   

    // 物件私有屬性 : 嚴禁其他模組直接使用私有屬性

    //

    int private_att1 ;

    int private_att2 ;

} ;

 

// 定義新的型別

//

typedef  MY_OBJ  my_object

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

// 類別屬性(class or static attribute) : 使用全域變數模擬

//         - 定義在my_object.c中

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

extern int my_obj_class_attr1 ;

extern int my_obj_class_attr2 ;

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

// 公用方法(public method)宣告

//         - function name必須可以很容易辨別這是某個物件的method

//         - 呼叫者必須主動傳入物件的指標

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~      

    // constructor –

// C語言不能有同名的函數，只能利用以下技巧實現C++中的同名方法

    //

int my_obj_constructor_1(MY_OBJ * the_object, int parameter1) ;

void my_obj_constructor_2(MY_OBJ * the_object) ;

 

// de-constructor

//

void my_obj_deconstructor(MY_OBJ * the_object) ;

 

    // 一般公用方法宣告

    //

        void my_obj_public_method1(MY_OBJ * the_object) ;

 

    // 外部程式必須透過使用物件的公用方法來操作物件的私有屬性

//

int my_obj_get_private_att1(MY_OBJ * the_object) ;

C-3 物件實例(instance)

 

C語言無法自動呼叫物件的constructor與de-constructor，必須由程式設計師視需要自行呼叫之。e譯器產生的者醒你寫程式在其他的程式模組中可以如以下實例使用這個物件：e譯器產生的者醒你寫程式

 

#include <my_obj.h>

 

void other_function(void)

{

 MY_OBJ A ;

 MY_OBJ B ;   

 

 int result ;

   

 // 使用者自行呼叫constructor 

 //

 result = my_obj_constructor_1(&A, 1) ;

 my_obj_constructor_2(&B) ;

 

 // 只允許操作公用屬性與公用方法

 //

 A.public_att1 = 0 ;

 B.public_att2 = A.public_att2 ;

 

// 物件方法調用 (invocation)

//

 my_obj_public_method1(&A) ;

 ...

 

 // 使用者自行呼叫deconstructor    

 //

 my_obj_deconstructor(&A) ;

 my_obj_deconstructor(&B) ;

 

}

C-5 基本類別定義

 

所有“類別(class)資料結構”定義時的第一個屬性，一定要是一個基本類別，程式才能e譯器產生的者醒你寫程式判斷傳入的物件指標究竟是指向哪一種物件，這樣才有辦法用C語言的轉型功能來實現物件導向觀念中的繼承以及多態性 (Polymorphism)。

 

基本類別的定義如下：

 

#define BASE_OBJ_VALID_FLAG 0xABCDEF

 

struct base_class

{

    // 以下屬性都是私有屬性，其他程式必須透過公用方法才能操作之          

    //

 

    // 這個flag如果不是特定值(e.g.0XABCDEF), 則表示不是合法的object

    //

    int valid_obj_flag ;

   

    // 系統中可為每一個類別編號, 依此欄位可得知傳入的物件為何

    //

    int object_type ;

   

    // 意義同object_type, 用字串來表示物件的類型

    //

    char object_name[16] ;     

   

    // 依據不同系統的需求, 系統設計者可以增加base-class的欄位

    //

} ;

 

typedef  OBJECT  base_class

 

 

 

C-6 基本類別的公用方法

 

基本類別的公用方法如下，就是要用來操作基本類別的私有屬性：

 

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

// Public Method定義

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

// 判斷傳入的物件指標是否為合法的物件

//

int base_obj_check_invalid_flag(OBJECT * the_object) ;

{

    if(the_object->valid_obj_flag == BASE_OBJ_VALID_FLAG)

            return true ;

    else

            return false ;

}

 

 

// 取得物件的型別

//

int base_obj_get_object_type(OBJECT * the_object) ;

{

    if(base_obj_check_invalid_flag(the_object == true)

            return the_object->object_type ;

    else

            return 0 ;       //    不合法的物件指標

}

 

 

 

// 取得物件的名稱

//

char * base_obj_get_object_name(OBJECT * the_object) ;

{

 

    if(base_obj_check_invalid_flag(the_object == true)

            return the_object->object_name ;

    else

            return NULL ;       //    不合法的物件指標

 

 

}

 

 

C-7 物件類型判斷範例

 

參考以下的實例就可以了解基本類別的意義：

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~// 列印傳入物件之屬性

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~void show_object_informtion(OBJECT * the_obj)

{

 // 判斷是否為合法物件

 //

 if( base_obj_check_invalid_flag(the_obj) == false)

 {

           show_string("this is not a valid object") ;

           return ; 

 }

 

 // 列印物件型態

 //

 switch( base_obj_get_object_type(the_object) )

 {

           case OBJ_TYPE_1:

                   show_string("this is object-1") ;

                   break ;

  

          case OBJ_TYPE_2:

                   show_string("this is object-2") ;

                   break ;

 

           default:

                   show_string("this is others object") ;

                   break ;

 }

   

// 列印物件名稱

 //

 show_string("object name is :") ;

 show_string(base_obj_get_object_name(the_object)) ;

   

}

 

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~//    測試程式

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~void test_program(void)

{

 OBJECT_TYPE_1 A ;

 OBJECT_TYPE_2 B ;

  

 // 執行物件A , B的contructor

 ...

 ...

 

 // 重點:   轉型為基本類別 - 因為所有物件的第一個屬性都是OBJECT

 //               所以可以用這樣的技巧

 //

 show_object_informtion((OBJECT *)(&A) ) ;

 

 // 所有的物件都可以轉成基本類別

 //

 show_object_informtion((OBJECT *)(&B) ) ;

   

   

}  

 

C-8 繼承(Inheritance)

 

用C語言實現繼承的技巧與基礎類別的觀念差不多，就是用到C語言的轉型：只要兒子類別的第一個屬性就是爸爸類別，則兒子物件轉型為爸爸物件後，可以使用爸爸物件的所有屬性與方法。

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

// 父親類別定義

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

struct parent_obj

{

 OBJECT base ;

 

 int papa_attr1 ;

 int papa_attr2 ;

 

} ;

 

typedef PAPA_OBJ struct parent_obj ;

 

// 父親類別的公用方法

//

void papa_public_method1(PAPA_OBJ * papa) ;

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~// 兒子類別定義

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

struct child_obj

{

 PAPA_OBJ papa ;           // 繼承自父親類別的屬性

 

 int child_attr1 ;

 int child_attr2 ;

   

} ;

 

typedef CHILD_OBJ struct child_obj ;

 

 

// 兒子類別的公用方法

//

void child_public_method1(CHILD_OBJ * papa) ;

 

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~// 測試程式

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

void test_program(void)

{

   CHILD_OBJ child ;

   PAPA_OBJ * pointer_to_my_parent_obj ;

 

// 操作兒子物件的公用方法與屬性

//

child.child_attr2 = 1 ;

child_public_method1(&child) ;     

       

// 轉型成爸爸物件

//

pointer_to_my_parent_obj = (PAPA_OBJ *)(&child) ;

       

// 操作爸爸物件的公用方法與屬性

//

pointer_to_my_parent_obj->papa_attr1 = 1 ;

papa_public_method1(pointer_to_my_parent_obj) ;

 

 

// 另一種操作爸爸物件的公用方法與屬性的寫法

//

child.papa.papa_attr1 = 1 ;

papa_public_method1((PAPA_OBJ *)(&child)) ;

}

 

 

C-9 多態性 (Polymorphism)

 

物件導向觀念中的polymorphism簡單的說：就是同一個function可以根據傳入的物件類型不同作不同的事情，在物件導向語言中是用overloading語法達成的，但在C語言中不允許相同名稱的function。我們基於繼承的範例來說明如何用C語言來模擬polymorphism的觀念：

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~// 兒子2類別定義

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

struct child_obj2

{

 PAPA_OBJ papa ;           // 繼承自父親類別的屬性

 

 int child2_attr1 ;

 int child2_attr2 ;

   

} ;

 

typedef CHILD_OBJ2 struct child_obj2 ;

 

// 兒子類別2的公用方法

//

void child2_public_method1(CHILD_OBJ * papa) ;

 

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~// 可處理多種類別的函數

//         - 傳入的參數是基本物件的指標

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

int papa_obj_polymorphism_method(OBJECT * kind_of_papa_obj)

{

 

// 判斷是否為合法物件

 //

 if( base_obj_check_invalid_flag(kind_of_papa_obj) == false)    

           return false ;

 

 // 判斷物件類型，根據不同的類型，呼叫該物件的public method

 //              

 switch( base_obj_get_object_type(kind_of_papa_obj) )

 {

            case OBJ_TYPE_PAPA:

                   papa_public_method1((PAPA_OBJ *)kind_of_papa_obj) ;

                   break ;

 

          case OBJ_TYPE_CHILD1:

                   child_public_method1((CHILD_OBJ *)kind_of_papa_obj) ;

                   break ;

  

          case OBJ_TYPE_CHILD2:

                    child2_public_method1((CHILD_OBJ2 *)kind_of_papa_obj) ;

                   break ;

 

           default:

                   return false ;         // 傳入錯誤的object

}

 

 return true ;           

   

}

 

 

 

 

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~// 測試程式

//~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

void test_program2(void)

{

 PAPA_OBJ      A ;

 CHILD_OBJ     B ;

 CHILD_OBJ2 C ;

 

 OTHER_OBJ    D ;

 

 // 執行各個物件的constructor

 //

 ...

 

  // 可以用同一個function處理不同的物件

 //

 papa_obj_polymorphism_method((OBJECT *)(&A)) ;

 papa_obj_polymorphism_method((OBJECT *)(&B)) ;

 papa_obj_polymorphism_method((OBJECT *)(&C)) ;

 papa_obj_polymorphism_method((OBJECT *)(&D)) ;
